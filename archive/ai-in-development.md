```typescript
// A2A Ğ´Ğ¾Ğ·Ğ²Ğ¾Ğ»ÑÑ” Ğ°Ğ³ĞµĞ½Ñ‚Ğ°Ğ¼ Ğ²Ñ–Ğ´ Ñ€Ñ–Ğ·Ğ½Ğ¸Ñ… Ğ¿Ñ€Ğ¾Ğ²Ğ°Ğ¹Ğ´ĞµÑ€Ñ–Ğ² ÑĞ¿Ñ–Ğ»ĞºÑƒĞ²Ğ°Ñ‚Ğ¸ÑÑŒ

    action: string;
    parameters: Record<string, unknown>;

  provider: 'amadeus',
  capabilities: ['search_flights', 'book_flight']
});

const hotelAgent = new A2AAgent({
  provider: 'booking',
  capabilities: ['search_hotels', 'book_hotel']
});

orchestrator.on('travel_request', async (request) => {
  // ĞŸĞ°Ñ€Ğ°Ğ»ĞµĞ»ÑŒĞ½Ğ¸Ğ¹ Ğ·Ğ°Ğ¿Ğ¸Ñ‚ Ğ´Ğ¾ Ñ€Ñ–Ğ·Ğ½Ğ¸Ñ… Ğ°Ğ³ĞµĞ½Ñ‚Ñ–Ğ²
  const [flights, hotels] = await Promise.all([
    flightAgent.send({ action: 'search_flights', parameters: request }),
    hotelAgent.send({ action: 'search_hotels', parameters: request })
  ]);
  
  return { flights, hotels };
});
```

## ĞŸÑ€Ğ°ĞºÑ‚Ğ¸Ñ‡Ğ½Ğ¸Ğ¹ Ğ¿Ñ€Ğ¸ĞºĞ»Ğ°Ğ´: Documentation Bot

```typescript
// doc-bot.ts
import { generateText, embed, embedMany } from 'ai';
import { openai } from '@ai-sdk/openai';
import { RecursiveCharacterTextSplitter } from 'langchain/text_splitter';

class DocumentationBot {
  private vectorStore: VectorStore;
  
  async indexDocumentation(docsPath: string) {
    // 1. Ğ§Ğ¸Ñ‚Ğ°Ñ”Ğ¼Ğ¾ Ğ²ÑÑ– markdown Ñ„Ğ°Ğ¹Ğ»Ğ¸
    const files = glob.sync(`${docsPath}/**/*.md`);
    
    // 2. ĞŸĞ°Ñ€ÑĞ¸Ğ¼Ğ¾ Ñ– Ñ‡Ğ°Ğ½ĞºÑƒÑ”Ğ¼Ğ¾
    const splitter = new RecursiveCharacterTextSplitter({
      chunkSize: 500,
      chunkOverlap: 50,
      separators: ['\n## ', '\n### ', '\n\n', '\n']
    });
    
    const chunks = [];
    for (const file of files) {
      const content = fs.readFileSync(file, 'utf-8');
      const fileChunks = await splitter.createDocuments(
        [content],
        [{ source: file }]
      );
      chunks.push(...fileChunks);
    }
    
    // 3. Ğ•Ğ¼Ğ±ĞµĞ´Ğ¸Ğ¼Ğ¾
    const { embeddings } = await embedMany({
      model: openai.embedding('text-embedding-3-small'),
      values: chunks.map(c => c.pageContent)
    });
    
    // 4. Ğ—Ğ±ĞµÑ€Ñ–Ğ³Ğ°Ñ”Ğ¼Ğ¾
    for (let i = 0; i < chunks.length; i++) {
      await this.vectorStore.add({
        id: `chunk_${i}`,
        embedding: embeddings[i],
        content: chunks[i].pageContent,
        metadata: chunks[i].metadata
      });
    }
    
    console.log(`âœ… Indexed ${chunks.length} chunks from ${files.length} files`);
  }
  
  async answer(question: string): Promise<string> {
    // 1. Embed question
    const { embedding } = await embed({
      model: openai.embedding('text-embedding-3-small'),
      value: question
    });
    
    // 2. Search
    const results = await this.vectorStore.search(embedding, 5);
    
    // 3. Filter by relevance
    const relevant = results.filter(r => r.score > 0.7);
    
    if (relevant.length === 0) {
      return "I couldn't find relevant information in the documentation.";
    }
    
    // 4. Build context
    const context = relevant
      .map(r => `[Source: ${r.metadata.source}]\n${r.content}`)
      .join('\n\n---\n\n');
    
    // 5. Generate answer
    const { text } = await generateText({
      model: openai('gpt-4o'),
      system: `You are a documentation assistant. Answer based ONLY on the provided context.
      If the answer is not in the context, say so.
      Always cite your sources.`,
      prompt: `Context:\n${context}\n\nQuestion: ${question}`
    });
    
    return text;
  }
}

// Ğ’Ğ¸ĞºĞ¾Ñ€Ğ¸ÑÑ‚Ğ°Ğ½Ğ½Ñ

await bot.indexDocumentation('./docs');

const answer = await bot.answer('How do I configure authentication?');
console.log(answer);
```
